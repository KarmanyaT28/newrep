<script>
    var intakeFormId = "{{ intake_form_id }}";
    
    function navigateToForm(formNumber) {
        var url = '';
        switch(formNumber) {
            case 1:
                url = `/form/form/${intakeFormId}/application_details/`;
                break;
            case 2:
                url = `/form/form/${intakeFormId}/secure_role/`;
                break;
            case 3:
                url = `/form/form/${intakeFormId}/role_modification/`;
                break;
            case 4:
                url = `/form/form/${intakeFormId}/role_access_form/`;
                break;
            case 5:
                url = `/form/form/${intakeFormId}/data_req/`;
                break;
            case 6:
                url = `/form/form/${intakeFormId}/host_req/`;
                break;
            case 7:
                url = `/form/form/${intakeFormId}/group_req/`;
                break;
            case 8:
                url = `/form/form/${intakeFormId}/other_req/`;
                break;
            case 9:
                url = `/form/form/${intakeFormId}/third_party/`;
                break;
        }
        window.location.href = url;
    }

    // Highlight the active button based on the current page
    window.onload = function() {
        var currentForm = {{ form_number }};  // Pass the current form number from your backend
        document.getElementById('btn-' + currentForm).classList.add('active-button');
    };
</script>
