    <!-- Navigation Buttons -->
    <div>
        <button id="btn-1" onclick="navigateToForm(1)">Application Details</button>
        <button id="btn-2" onclick="navigateToForm(2)">Secure Role</button>
        <button id="btn-3" onclick="navigateToForm(3)">Role Modification</button>
        <button id="btn-4" onclick="navigateToForm(4)">Data Requirements</button>
        <button id="btn-5" onclick="navigateToForm(5)">Host Requirements</button>
        <button id="btn-6" onclick="navigateToForm(6)">Group Requirements</button>
    </div>

    <!-- Form Example -->
    <form method="POST" onsubmit="markFormComplete(1)">
        <!-- Form fields here -->
        <button type="submit" class="btn btn-primary">Save and Continue</button>
    </form>

    <script>
        var intakeFormId = "{{ intake_form_id }}";  // Assuming this is passed from Django

        function navigateToForm(formNumber) {
            var url = '';
            switch (formNumber) {
                case 1:
                    url = `/form/${intakeFormId}/application/`;
                    break;
                case 2:
                    url = `/form/${intakeFormId}/secure-role/`;
                    break;
                case 3:
                    url = `/form/${intakeFormId}/role-modification/`;
                    break;
                case 4:
                    url = `/form/${intakeFormId}/data-req/`;
                    break;
                case 5:
                    url = `/form/${intakeFormId}/host-req/`;
                    break;
                case 6:
                    url = `/form/${intakeFormId}/group-req/`;
                    break;
            }
            window.location.href = url;
        }

        function markFormComplete(formNumber) {
            // Store the completed form in localStorage
            localStorage.setItem(`form_${formNumber}_completed`, true);
            updateButtonColors();
        }

        function updateButtonColors() {
            for (let i = 1; i <= 6; i++) {
                if (localStorage.getItem(`form_${i}_completed`)) {
                    document.getElementById(`btn-${i}`).classList.add('completed');
                }
            }
        }

        window.onload = function() {
            updateButtonColors();  // Check completion status on page load
        }
    </script>
