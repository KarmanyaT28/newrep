    <script>
        function validateForm() {
            const memberId = document.getElementById('member_identifier').value;
            const memberType = document.getElementById('member_identifier_type').value;
            const databaseName = document.getElementById('database_name').value;
            const restrictedIds = document.getElementById('restricted_to_ids').value;

            if (memberId !== '' && memberType === '') {
                alert('Please select a Member Identifier Type.');
                return false;
            }

            if ((memberType.includes('db role')) && databaseName === '') {
                alert('Please provide a Database Name for the selected Database Role.');
                return false;
            }

            if (memberType === 'restricted to ids' && restrictedIds === '') {
                alert('Please provide Restricted to IDs.');
                return false;
            }

            return true;
        }

        function toggleFields() {
            const memberId = document.getElementById('member_identifier').value;
            const memberType = document.getElementById('member_identifier_type').value;

            document.getElementById('member_identifier_type').disabled = (memberId === '');

            if (memberType.includes('db role')) {
                document.getElementById('database_name').disabled = false;
                document.getElementById('restricted_to_ids').disabled = true;
            } else if (memberType === 'restricted to ids') {
                document.getElementById('database_name').disabled = true;
                document.getElementById('restricted_to_ids').disabled = false;
            } else {
                document.getElementById('database_name').disabled = true;
                document.getElementById('restricted_to_ids').disabled = true;
            }
        }
    </script>
